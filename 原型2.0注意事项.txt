					}, {

						iconCls: 'icon-prev'

					}]
				})
				//禁止整个页面的右键
				Ext.getDoc().on("contextmenu", function(e) {
					e.stopEvent();
				});
				//注册右键
				Ext.getBody().on("contextmenu", function(e) {
					menu.showAt(e.getPoint().x, e.getPoint().y); //显示在当前位置 
				});
